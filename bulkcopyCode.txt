private IConfiguration _configuration;
        public DistrictController(IConfiguration configuration)
        {
            _configuration = configuration;
        }

        [HttpGet]
        public IActionResult JustMakeRequest() 
        {
            DataTable dt = new DataTable();
            //dt.Columns.Add("ID",typeof(int));
            dt.Columns.Add("STCode", typeof(int));
            dt.Columns.Add("StateName", typeof(string));
            dt.Columns.Add("DTCode", typeof(int));
            dt.Columns.Add("DistrictName", typeof(string));
            dt.Columns.Add("SDTCode", typeof(int));
            dt.Columns.Add("SubDistrictName", typeof(string));
            dt.Columns.Add("TownCode", typeof(int));
            dt.Columns.Add("AreaName", typeof(string));

            string path = Path.GetFullPath("C:/Users/Amit Limbasiya/Downloads/gujarat.csv");
            string csvData = System.IO.File.ReadAllText(path);
            foreach (string csvRow in csvData.Split("\n"))
            {
                if (!string.IsNullOrEmpty(csvRow))
                {
                    //Adding each row into datatable  
                    dt.Rows.Add();
                    int count = 0;
                    foreach (string FileRec in csvRow.Split(','))
                    {
                        Console.WriteLine(FileRec);
                        if(count%2==0)
                            dt.Rows[dt.Rows.Count - 1][count] = int.Parse(FileRec);
                        else
                            dt.Rows[dt.Rows.Count - 1][count] = FileRec;
                        count++;
                    }
                }
            }
            var con = new SqlConnection(_configuration.GetConnectionString("Default"));
            SqlBulkCopy objbulk = new SqlBulkCopy(con);
            //assigning Destination table name    
            objbulk.DestinationTableName = "Districts";
            //Mapping Table column    
            objbulk.ColumnMappings.Add(0, "STCode");
            objbulk.ColumnMappings.Add(1, "StateName");
            objbulk.ColumnMappings.Add(2, "DTCode");
            objbulk.ColumnMappings.Add(3, "DistrictName");
            objbulk.ColumnMappings.Add(4, "SDTCode");
            objbulk.ColumnMappings.Add(5, "SubDistrictName");
            objbulk.ColumnMappings.Add(6, "TownCode");
            objbulk.ColumnMappings.Add(7, "AreaName");
            //inserting Datatable Records to DataBase    
            con.Open();
            objbulk.WriteToServer(dt);
            con.Close();
            return BadRequest("Data entry not successful");
        }
